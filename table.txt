  CREATE TABLE weather(
    cityid INTEGER ,
    name VARCHAR ,
    country VARCHAR ,
    dt INTEGER ,
    temp DECIMAL,
    description VARCHAR ,
    icon VARCHAR ,
    wind DECIMAL,
    dt_text DATE,
    PRIMARY KEY (cityid, dt)
  );
